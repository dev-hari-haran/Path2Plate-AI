Arduino — Bluetooth HC-05 / HC-06 Code

▸ Arduino — HC-05/06 Bluetooth Motor Control
#include <SoftwareSerial.h>

// Bluetooth module pins (connect HC-05 TX → pin 10, RX → pin 11)
SoftwareSerial BT(10, 11); // RX, TX

// Motor driver pins (L298N or similar)
// Left motor
#define IN1 2
#define IN2 3
#define ENA 5   // PWM speed

// Right motor
#define IN3 4
#define IN4 7
#define ENB 6   // PWM speed

int motorSpeed = 180;  // 0-255 PWM speed

void setup() {
  Serial.begin(9600);
  BT.begin(9600);  // HC-05 default baud

  pinMode(IN1,OUTPUT); pinMode(IN2,OUTPUT); pinMode(ENA,OUTPUT);
  pinMode(IN3,OUTPUT); pinMode(IN4,OUTPUT); pinMode(ENB,OUTPUT);

  Serial.println("Bluetooth Ready");
}

void loop() {
  if (BT.available()) {
    String cmd = BT.readStringUntil('\n');
    cmd.trim();
    Serial.println("CMD: " + cmd);
    handleCommand(cmd);
  }
}

void handleCommand(String cmd) {
  // Supports: FORWARD, BACKWARD, LEFT, RIGHT, STOP
  // Also supports JSON: {"cmd":"FORWARD"}
  // Also supports single char: F, B, L, R, S

  if (cmd == "FORWARD" || cmd == "F" || cmd.indexOf("FORWARD") >= 0) {
    moveForward();
  } else if (cmd == "BACKWARD" || cmd == "B" || cmd.indexOf("BACKWARD") >= 0) {
    moveBackward();
  } else if (cmd == "LEFT" || cmd == "L" || cmd.indexOf("LEFT") >= 0) {
    turnLeft();
  } else if (cmd == "RIGHT" || cmd == "R" || cmd.indexOf("RIGHT") >= 0) {
    turnRight();
  } else if (cmd == "STOP" || cmd == "S" || cmd.indexOf("STOP") >= 0) {
    stopMotors();
  }
}

void moveForward() {
  digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW);
  digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW);
  analogWrite(ENA, motorSpeed); analogWrite(ENB, motorSpeed);
}

void moveBackward() {
  digitalWrite(IN1, LOW); digitalWrite(IN2, HIGH);
  digitalWrite(IN3, LOW); digitalWrite(IN4, HIGH);
  analogWrite(ENA, motorSpeed); analogWrite(ENB, motorSpeed);
}

void turnLeft() {
  digitalWrite(IN1, LOW);  digitalWrite(IN2, HIGH);
  digitalWrite(IN3, HIGH); digitalWrite(IN4, LOW);
  analogWrite(ENA, motorSpeed); analogWrite(ENB, motorSpeed);
}

void turnRight() {
  digitalWrite(IN1, HIGH); digitalWrite(IN2, LOW);
  digitalWrite(IN3, LOW);  digitalWrite(IN4, HIGH);
  analogWrite(ENA, motorSpeed); analogWrite(ENB, motorSpeed);
}

void stopMotors() {
  digitalWrite(IN1,LOW); digitalWrite(IN2,LOW);
  digitalWrite(IN3,LOW); digitalWrite(IN4,LOW);
  analogWrite(ENA, 0);   analogWrite(ENB, 0);
}
▸ HC-05 / HC-06 Wiring
HC-05/HC-06     Arduino Uno/Nano
─────────────────────────────────
VCC         →   5V
GND         →   GND
TXD         →   Pin 10 (SoftwareSerial RX)
RXD         →   Pin 11 (SoftwareSerial TX)
                (use 1kΩ + 2kΩ voltage divider on RXD if 5V board)

L298N Motor Driver
─────────────────────────────────
IN1         →   Arduino Pin 2
IN2         →   Arduino Pin 3
ENA         →   Arduino Pin 5 (PWM)
IN3         →   Arduino Pin 4
IN4         →   Arduino Pin 7
ENB         →   Arduino Pin 6 (PWM)
12V         →   Battery +
GND         →   Battery − & Arduino GND

Pairing Steps
─────────────────────────────────
1. Power on HC-05 (LED blinks fast = not paired)
2. Go to System Bluetooth Settings on your device
3. Pair with "HC-05" or "HC-06" — default PIN: 1234 or 0000
4. LED blinks slow = paired
5. Click "CONNECT BLUETOOTH" in the sim
▸ Command Reference
The simulation sends commands based on selected format:

FORMAT: Plain Text
  FORWARD    — move forward
  BACKWARD   — move backward
  LEFT       — turn left
  RIGHT      — turn right
  STOP       — stop all motors

FORMAT: JSON
  {"cmd":"FORWARD"}
  {"cmd":"BACKWARD"}
  {"cmd":"LEFT"}
  {"cmd":"RIGHT"}
  {"cmd":"STOP"}

FORMAT: Single Char
  F  — FORWARD
  B  — BACKWARD
  L  — LEFT
  R  — RIGHT
  S  — STOP

Duration Control
  The sim sends FORWARD/BACKWARD for [FB duration] seconds,
  LEFT/RIGHT for [LR duration] seconds, then auto-sends STOP.
  Tune these values to match your robot's physical movement
  so 1 grid cell ≈ 1 step of the bot.

Movement Timing Example:
  FB Duration = 0.5s  →  bot moves ~20cm forward
  LR Duration = 0.4s  →  bot rotates ~90°
  Stop Delay  = 0.1s  →  gap between commands
